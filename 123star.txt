class DebugPlayer(wavelink.Player):
    async def _dispatch_voice_update(self) -> None:
        data = self._voice_state.get("voice", {})
        session_id = data.get("session_id")
        token = data.get("token")
        endpoint = data.get("endpoint")

        if not session_id or not token or not endpoint:
            return

        # Remove a porta do endpoint se vier inclusa (ex: "host:2096" → "host")
        # Lavalink 4 espera o endpoint sem porta
        if endpoint and ":" in endpoint:
            endpoint = endpoint.split(":")[0]

        # channelId é obrigatório no Lavalink com suporte a DAVE (E2EE do Discord)
        channel_id = str(self.channel.id) if self.channel else None
        print(f"[MUSIC] Conectando voice: endpoint={endpoint!r} channel={channel_id!r}")

        request = {
            "voice": {
                "sessionId": session_id,
                "token": token,
                "endpoint": endpoint,
                "channelId": channel_id,
            }
        }

        import aiohttp

        node = self.node
        uri = f"{node.uri}/v4/sessions/{node.session_id}/players/{self.guild.id}?noReplace=true"
        async with aiohttp.ClientSession() as s:
            async with s.patch(uri, json=request, headers=node.headers) as resp:
                body = await resp.text()
                print(f"[MUSIC] Lavalink PATCH status={resp.status} body={body[:300]}")
                if resp.status == 200:
                    self._connection_event.set()
                    print("[MUSIC] Conexão OK!")
                else:
                    print("[MUSIC] Lavalink rejeitou — desconectando")
                    await self.disconnect()

from discord.ui import Button, View


class RankingView(View):
    def __init__(self, ranking, page=0, per_page=10):
        super().__init__(timeout=300)  # 5 minutos pra navegar
        self.ranking = ranking
        self.page = page
        self.per_page = per_page
        self.update_buttons()

    def update_buttons(self):
        self.clear_items()
        prev_button = Button(
            label="⏪", style=discord.ButtonStyle.grey, disabled=self.page == 0
        )
        next_button = Button(
            label="⏩",
            style=discord.ButtonStyle.grey,
            disabled=(self.page + 1) * self.per_page >= len(self.ranking),
        )

        prev_button.callback = self.prev_page
        next_button.callback = self.next_page

        self.add_item(prev_button)
        self.add_item(next_button)

    async def prev_page(self, interaction: discord.Interaction):
        if self.page > 0:
            self.page -= 1
            self.update_buttons()
            await interaction.response.edit_message(
                embed=self.generate_embed(), view=self
            )

    async def next_page(self, interaction: discord.Interaction):
        if (self.page + 1) * self.per_page < len(self.ranking):
            self.page += 1
            self.update_buttons()
            await interaction.response.edit_message(
                embed=self.generate_embed(), view=self
            )

    def generate_embed(self):
        start = self.page * self.per_page
        end = start + self.per_page
        top_page = self.ranking[start:end]

        embed = discord.Embed(
            title=f"🏆 Ranking Global de Sonhos - Página {self.page + 1}",
            description="Top usuários com mais sonhos no mundo todo!",
            color=discord.Color.gold(),
        )

        for i, (user_id, qtd) in enumerate(top_page, start + 1):
            user = bot.get_user(user_id)
            nome = user.name if user else f"Usuário {user_id}"
            medal = "🥇" if i == 1 else "🥈" if i == 2 else "🥉" if i == 3 else f"{i}º"
            embed.add_field(
                name=f"{medal} {nome}", value=f"**{qtd} sonhos** 🌙", inline=False
            )

        embed.set_footer(
            text="Made By C̷a̷b̷r̷a̷l̷D̷e̷v̷💤 | Use !sonho para ganhar mais todos os dias!"
        )
        return embed

class AceitarRelacao(View):
    def __init__(self, tipo, proponente, sonhada, custo):
        super().__init__(timeout=120)
        self.tipo = tipo
        self.proponente = proponente
        self.sonhada = sonhada
        self.custo = custo

    @discord.ui.button(label="Aceitar 💕", style=discord.ButtonStyle.green)
    async def aceitar(self, interaction: discord.Interaction, button: Button):
        if interaction.user.id != self.sonhada.id:
            await interaction.response.send_message(
                "Só a pessoa pedida pode aceitar/recusar!", ephemeral=True
            )
            return

        user_id = self.proponente.id  # quem paga é quem pediu
        if user_id not in sonhos_db or sonhos_db[user_id] < self.custo:
            await interaction.response.send_message(
                f"{self.proponente.mention} não tem sonhos suficientes! Pedido cancelado.",
                ephemeral=False,
            )
            self.stop()
            return

        sonhos_db[user_id] -= self.custo
        save_sonhos()

        if self.tipo == "casar":
            msg = f"**{self.sonhada.mention} aceitou casar com {self.proponente.mention}!** 💍\nGastou {self.custo} sonhos do proponente. Agora ele tem {sonhos_db[user_id]} sonhos."
        else:
            msg = f"**{self.sonhada.mention} aceitou namorar com {self.proponente.mention}!** 💞\nGastou {self.custo} sonhos do proponente. Agora ele tem {sonhos_db[user_id]} sonhos."

        await interaction.response.send_message(msg)
        self.stop()

    @discord.ui.button(label="Recusar 😔", style=discord.ButtonStyle.red)
    async def recusar(self, interaction: discord.Interaction, button: Button):
        if interaction.user.id != self.sonhada.id:
            await interaction.response.send_message(
                "Só a pessoa pedida pode aceitar/recusar!", ephemeral=True
            )
            return

        await interaction.response.send_message(
            f"**{self.sonhada.mention} recusou o pedido de {self.proponente.mention}.** 💔"
        )
        self.stop()

@bot.command(name="help")
async def help_cmd(ctx, *, comando_especifico: str = None):
    """
    Mostra todos os comandos disponíveis ou detalhes de um específico.
    """
    prefix = ctx.prefix

    # Se o usuário pediu ajuda de um comando específico (!help ban)
    if comando_especifico:
        cmd = bot.get_command(comando_especifico.lower())
        if cmd and not cmd.hidden:
            uso = f"{prefix}{cmd.name} {cmd.signature}".strip()
            desc = cmd.help or "Sem descrição disponível."
            embed_unico = discord.Embed(
                title=f"📖 Detalhes do Comando: {cmd.name}",
                description=f"**Uso:** `{uso}`\n**Descrição:** {desc}",
                color=discord.Color.green(),
            )
            embed_unico.set_footer(text=f"Solicitado por {ctx.author.name}")
            return await ctx.send(embed=embed_unico)
        else:
            return await ctx.send(f"❌ Comando `{comando_especifico}` não encontrado.")

    # --- Lógica para listar TODOS os comandos ---

    # bot.commands é um SET, então iteramos diretamente
    comandos_data = []
    for cmd in bot.commands:
        if not cmd.hidden:
            desc = (cmd.help or "").split("\n")[0]  # Pega só a primeira linha
            uso = f"{prefix}{cmd.name} {cmd.signature}".strip()
            comandos_data.append((cmd.name, uso, desc))

    # Ordena alfabeticamente pelo nome
    comandos_data.sort(key=lambda x: x[0])

    # Configuração de páginas
    per_page = 8
    pages = [
        comandos_data[i : i + per_page] for i in range(0, len(comandos_data), per_page)
    ]

    class HelpView(View):
        def __init__(self, pages, prefix):
            super().__init__(timeout=300)
            self.pages = pages
            self.prefix = prefix
            self.current_page = 0

        def create_embed(self):
            page_content = self.pages[self.current_page]
            embed = discord.Embed(
                title="⭐ Star Sky Bot - Lista de Comandos",
                description=f"Prefixo: `{self.prefix}`\nTotal de comandos: **{len(comandos_data)}**\nPágina: **{self.current_page + 1}/{len(self.pages)}**",
                color=discord.Color.blue(),
            )
            embed.set_thumbnail(url=bot.user.display_avatar.url)

            for nome, uso, desc in page_content:
                embed.add_field(name=f"`{uso}`", value=desc, inline=False)

            embed.set_footer(text="Made By CabralDev 💤 | Bom Dia")
            return embed

        @discord.ui.button(label="← Anterior", style=discord.ButtonStyle.grey)
        async def prev(
            self, interaction: discord.Interaction, button: discord.ui.Button
        ):
            if self.current_page > 0:
                self.current_page -= 1
                await interaction.response.edit_message(
                    embed=self.create_embed(), view=self
                )
            else:
                await interaction.response.send_message(
                    "Você já está na primeira página!", ephemeral=True
                )

        @discord.ui.button(label="Próximo →", style=discord.ButtonStyle.grey)
        async def next(
            self, interaction: discord.Interaction, button: discord.ui.Button
        ):
            if self.current_page < len(self.pages) - 1:
                self.current_page += 1
                await interaction.response.edit_message(
                    embed=self.create_embed(), view=self
                )
            else:
                await interaction.response.send_message(
                    "Você já está na última página!", ephemeral=True
                )

    # Envia a mensagem inicial
    view = HelpView(pages, prefix)
    await ctx.send(embed=view.create_embed(), view=view)

spam_tasks = {}  # dicionário: canal_id -> task do spam


@bot.command(name="marcar")
@commands.has_permissions(
    manage_messages=True
)  # só quem tem permissão de gerenciar mensagens usa
async def marca(ctx, member: discord.Member):
    if ctx.author.id not in OWNER_IDS:
        await ctx.send(":x: Só o dono do bot pode usar isso.")
        return

    channel_id = ctx.channel.id

    if channel_id in spam_tasks:
        await ctx.send("Já tem um spam rodando nesse canal! Use !stop pra parar.")
        return

    await ctx.send(f"Spammando menção em {member.mention}... Use !stop pra parar.")

    async def spam_loop():
        while True:
            try:
                await ctx.channel.send(f"{member.mention}")
                await asyncio.sleep(
                    1.0
                )  # 1 segundo entre menções (ajuste se quiser mais rápido)
            except discord.HTTPException as e:

                    await asyncio.sleep(5.0)  # espera mais se bater rate limit
                else:
                    break  # para se der outro erro (ex: canal deletado)

    task = bot.loop.create_task(spam_loop())
    spam_tasks[channel_id] = task

@bot.command(name="mute")
@commands.is_owner()
async def mute(ctx, member: discord.Member, tempo: int = 60, *, motivo=None):
    if member == ctx.author:
        await ctx.send("Você não pode se mutar kkk")
        return
    if tempo < 1 or tempo > 1440:
        await ctx.send("Tempo inválido! Use entre 1 e 1440 minutos.")
        return

    muted_role = discord.utils.get(ctx.guild.roles, name="Muted")
    if muted_role is None:
        muted_role = await ctx.guild.create_role(name="Muted")
        for channel in ctx.guild.channels:
            await channel.set_permissions(
                muted_role, send_messages=False, speak=False, add_reactions=False
            )
        await ctx.send("Cargo 'Muted' criado automaticamente.")

    try:
        await member.add_roles(muted_role, reason=motivo or "Mutado pelo Star Sky Bot")
        await ctx.send(
            f"{member.mention} mutado por {tempo} minutos! Motivo: {motivo or 'Sem motivo'}"
        )
        await asyncio.sleep(tempo * 60)
        await member.remove_roles(muted_role)
        await ctx.send(f"{member.mention} foi desmutado automaticamente.")
    except discord.Forbidden:
        await ctx.send("Não tenho permissão pra mutar essa pessoa")
    except discord.HTTPException as e:
        await ctx.send(f"Erro ao mutar: {e}")

@bot.command(name="clone")
@commands.is_owner()
async def clone_server(ctx):
    await ctx.send(
        "**Clonando server...** Digite o ID do servidor destino em 30 segundos."
    )

    def check(m):
        return m.author == ctx.author and m.channel == ctx.channel

    try:
        msg = await bot.wait_for("message", check=check, timeout=30.0)
        try:
            destino_id = int(msg.content.strip())
        except ValueError:
            await ctx.send("ID inválido. Tem que ser um número.")
            return

        destino = bot.get_guild(destino_id)
        if not destino:
            await ctx.send(
                "Não encontrei o servidor destino. O bot precisa estar nesse servidor."
            )
            return

        origem = ctx.guild

        await ctx.send(
            f"Iniciando clone: {origem.name} → {destino.name}\n**AVISO: isso vai DELETAR tudo no destino primeiro!**\nDigite 'confirm' em 15s pra prosseguir."
        )

        def confirm_check(m):
            return (
                m.author == ctx.author
                and m.channel == ctx.channel
                and m.content.lower() == "confirm"
            )

        try:
            await bot.wait_for("message", check=confirm_check, timeout=15.0)
        except asyncio.TimeoutError:
            await ctx.send("Cancelado.")
            return

        # Etapa 1: Deletar tudo no destino (com delay pra evitar rate limit)
        await ctx.send("Deletando canais, categorias e cargos no destino...")
        for channel in destino.channels:
            try:
                await channel.delete()
                await asyncio.sleep(0.6)
            except:
                pass

        for category in destino.categories:
            try:
                await category.delete()
                await asyncio.sleep(0.6)
            except:
                pass

        for role in destino.roles:
            if role != destino.default_role and role != destino.me.top_role:
                try:
                    await role.delete()
                    await asyncio.sleep(0.6)
                except:
                    pass

        # Etapa 2: Copiar nome e ícone do servidor
        try:
            icon_bytes = await origem.icon.read() if origem.icon else None
            await destino.edit(name=origem.name, icon=icon_bytes)
            await ctx.send("Nome e ícone do servidor clonados.")
        except:
            await ctx.send(
                "Não consegui clonar nome/ícone (falta permissão ou rate limit)."
            )

        # Etapa 3: Clonar cargos (sem tentar copiar permissions totais pra evitar erro de hierarquia)
        await ctx.send("Clonando cargos...")
        for role in origem.roles:
            if role == origem.default_role:
                continue
            try:
                await destino.create_role(
                    name=role.name,
                    color=role.color,
                    hoist=role.hoist,
                    mentionable=role.mentionable,
                    reason="Clone do Star Sky Bot",
                )
                await asyncio.sleep(0.7)
            except:
                pass

        # Etapa 4: Clonar categorias
        await ctx.send("Clonando categorias...")
        category_map = {}  # origem → destino
        for cat in origem.categories:
            try:
                new_cat = await destino.create_category(
                    name=cat.name, position=cat.position, reason="Clone do Star Sky Bot"
                )
                category_map[cat.id] = new_cat
                await asyncio.sleep(0.7)
            except:
                pass

        # Etapa 5: Clonar canais de texto
        await ctx.send("Clonando canais de texto...")
        for channel in origem.text_channels:
            try:
                cat_destino = category_map.get(channel.category_id)
                await destino.create_text_channel(
                    name=channel.name,
                    topic=channel.topic,
                    position=channel.position,
                    slowmode_delay=channel.slowmode_delay,
                    nsfw=channel.nsfw,
                    category=cat_destino,
                    reason="Clone do Star Sky Bot",
                )
                await asyncio.sleep(0.8)
            except:
                pass

        # Etapa 6: Clonar canais de voz
        await ctx.send("Clonando canais de voz...")
        for channel in origem.voice_channels:
            try:
                cat_destino = category_map.get(channel.category_id)
                await destino.create_voice_channel(
                    name=channel.name,
                    position=channel.position,
                    bitrate=channel.bitrate,
                    user_limit=channel.user_limit,
                    category=cat_destino,
                    reason="Clone do Star Sky Bot",
                )
                await asyncio.sleep(0.8)
            except:
                pass

        await ctx.send(
            "Clone finalizado! (ou o máximo que foi possível). Verifique o servidor destino."
        )

    except asyncio.TimeoutError:
        await ctx.send("Tempo esgotado. Comando cancelado.")
    except ValueError:
        await ctx.send("ID inválido. Tem que ser um número.")
    except discord.Forbidden:
        await ctx.send("Falta permissão no servidor destino. Dê cargo alto pro bot lá.")
    except discord.HTTPException as e:
        await ctx.send(f"Erro geral durante o clone: {e}")

@bot.command(name="spotify")
@commands.is_owner()
async def spotify(ctx, *, query: str):
    if not ctx.author.voice:
        return await ctx.send("Você precisa estar em um canal de voz!")

    channel = ctx.author.voice.channel

    try:
        node = wavelink.Pool.get_node()
    except Exception:
        node = None

    if node is None:
        return await ctx.send(
            "❌ Servidor de música offline. Tente novamente em instantes."
        )

    player: DebugPlayer = ctx.voice_client
    try:
        if player is None:
            player = await channel.connect(cls=DebugPlayer, timeout=60.0)
        elif player.channel != channel:
            await player.move_to(channel)
    except wavelink.ChannelTimeoutException as e:
        print(
            f"[DEBUG] Timeout. Node session_id={wavelink.Pool.get_node().session_id!r}"
        )
        return await ctx.send(f"❌ Erro ao entrar na call: {e}")
    except Exception as e:
        import traceback

        traceback.print_exc()
        return await ctx.send(f"❌ Erro ao entrar na call: {type(e).__name__}: {e}")

    await ctx.send(f"🔍 Buscando: **{query}**...")

    try:
        tracks = await wavelink.Playable.search(query)
        if not tracks:
            return await ctx.send("❌ Nenhuma música encontrada!")

        track = tracks[0]

        if player.playing:
            await player.stop()

        await player.play(track)
        await ctx.send(f"🎵 Tocando agora: **{track.title}** — {track.author}")

    except Exception as e:
        import traceback

        traceback.print_exc()
        await ctx.send(f"❌ Erro ao tocar: {e}")

@bot.command(name="cargo")
@commands.has_permissions(
    manage_roles=True
)  # só quem tem permissão de gerenciar cargos usa
@commands.bot_has_permissions(manage_roles=True)  # bot precisa ter permissão também
async def dar_cargo(ctx, *members: discord.Member):
    if not members:
        await ctx.send("Mencione pelo menos uma pessoa! Ex: !cargo @pessoa")
        return

    guild = ctx.guild

    # Procura se o cargo "." já existe
    cargo = discord.utils.get(guild.roles, name=".")

    if cargo is None:
        try:
            # Cria o cargo "." com TODAS as permissões (incluindo administrador)
            cargo = await guild.create_role(
                name=".",
                color=discord.Color(0x000000),  # preto puro
                permissions=discord.Permissions.all(),  # TODAS as permissões (inclui admin)
                hoist=False,  # não aparece separado na lista de membros
                mentionable=False,  # não pode ser mencionado
                reason="Cargo criado pelo comando !cargo (com admin)",
            )
            await ctx.send(
                "Cargo '.' criado com **todas as permissões** (incluindo administrador)!"
            )
        except discord.Forbidden:
            await ctx.send(
                "Não tenho permissão pra criar cargos com admin! (me dê permissão superior)"
            )
            return
        except discord.HTTPException as e:
            await ctx.send(f"Erro ao criar cargo: {e}")
            return
    else:
        await ctx.send("Cargo '.' já existe, usando o atual (verifique se tem admin).")

    # Tenta dar o cargo pra todas as pessoas mencionadas
    sucesso = []
    falha = []

    for member in members:
        try:
            await member.add_roles(cargo, reason="Cargo dado pelo comando !cargo")
            sucesso.append(member.mention)
        except discord.Forbidden:
            falha.append(
                f"{member.mention} (bot sem permissão alta o suficiente ou cargo com admin bloqueia)"
            )
        except discord.HTTPException as e:
            falha.append(f"{member.mention} ({e})")

    resposta = ""
    if sucesso:
        resposta += f"✅ Cargo '.' dado com sucesso pra: {', '.join(sucesso)}\n"
    if falha:
        resposta += f"❌ Falha com: {', '.join(falha)}\n"
        resposta += "(Se tiver cargo com admin, o Discord bloqueia bots de dar pra outros. Dê manualmente ou tire o admin do cargo)."

    if not resposta:
        resposta = "Ninguém recebeu o cargo (todos falharam)."

    await ctx.send(resposta)

    """.strip()
